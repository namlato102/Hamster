453 {

462  for(k = 0; k < j; k++) /* hex dump */
463  printf("%02X ", *(p+k));
464  for(k = k; k < 16; k++) /* fill remainder if line was < 16 */
465  printf(" ");

469  putchar('.'); /* print controls etc. as dot! */
470  else

472
